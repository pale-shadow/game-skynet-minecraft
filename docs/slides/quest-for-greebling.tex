% Slide 2: The Distributed Architecture
\begin{frame}{System Topology and Node Roles}
	\begin{itemize}
		\item \textbf{Skynet:} AI orchestration, neural routing, and high-level agent management.
		\item \textbf{Stargate:} Control plane, handling MCP integration and inter-server bridging.
		\item \textbf{Chonk:} World-state execution layer running the live Minecraft instance.
		\item \textbf{Edge-T:} Telemetry collection and peripheral edge computing tasks.
	\end{itemize}
\end{frame}

% Slide 3: The Chonk Execution Layer
\begin{frame}{The Chonk Execution Layer}
	\begin{itemize}
		\item \textbf{Environment:} Debian 12 (Bookworm) and Java 21.
		\item \textbf{Core Engine:} Paper 1.21.11 / Forge integration.
		\item \textbf{Role:} Handles active game simulation and local physical mechanics.
		\item \textbf{Design Choice:} Fully isolated from heavy NPU/TPU AI orchestration to maintain high TPS and server stability.
	\end{itemize}
\end{frame}

% Slide 4: Physical Detection Mechanics
\begin{frame}{Physical Detection in World-Space}
	\begin{itemize}
		\item \textbf{Mechanism:} Detector rails capture the spatial movement of minecarts.
		\item \textbf{Signal Conditioning:} 1-tick repeaters stabilize redstone pulses during high-speed transits.
		\item \textbf{Triggering:} Stabilized pulses reliably activate adjacent Command Blocks.
		\item \textbf{Result:} Smoothly converts physical entity movement into discrete digital events.
	\end{itemize}
		\begin{center}\includegraphics[scale=0.07]{images/Gemini_Generated_Image_xz297bxz297bxz29.png}\end{center}
\end{frame}

% Slide 5: Telemetry Encapsulation
\begin{frame}{Telemetry Encapsulation}
	\begin{itemize}
		\item \textbf{Action:} Command Blocks execute targeted \texttt{tellraw} payloads.
		\item \textbf{Syntax:} \texttt{... run tellraw @a \{"text":"[TELEMETRY] NODE:CHONK EVENT:CART\_PASS"\}}
		\item \textbf{Rationale:} Bypasses standard JVM and "Server" prefix clutter in console logs.
		\item \textbf{Output:} Clean, standardized string injected directly into the RCON buffer.
	\end{itemize}
\end{frame}

% Slide 6: The Local Bridge Service
\begin{frame}{The Local Bridge Service}
	\begin{itemize}
		\item \textbf{Architecture:} Python/Go-based bridge script running locally on the Chonk host.
		\item \textbf{Monitoring:} Actively tails \texttt{logs/latest.log} and monitors the RCON stream.
		\item \textbf{Pattern Matching:} Filters for the \texttt{[TELEMETRY]} identifier.
		\item \textbf{Transformation:} Extracts metadata (NODE, EVENT) and encapsulates it into a network-routable JSON payload.
	\end{itemize}
\end{frame}

% Slide 7: Inter-Node Network Transit
\begin{frame}{Inter-Node Network Transit}
	\begin{itemize}
		\item \textbf{Protocol:} Asynchronous TCP/UDP socket communication.
		\item \textbf{Direction:} Forwards JSON payloads from Chonk directly to the Stargate host.
		\item \textbf{Performance:} Network transit occurs in an isolated thread.
		\item \textbf{Benefit:} Prevents network latency from causing "tick lag" on the live Minecraft server.
	\end{itemize}
\end{frame}

% Slide 8: Ingestion and MCP Integration
\begin{frame}{Stargate Ingestion \& MCP Integration}
	\begin{itemize}
		\item \textbf{Ingestion:} Stargate daemon listens asynchronously on port 5005.
		\item \textbf{Translation:} Decodes incoming JSON telemetry packets.
		\item \textbf{Action Mapping:} Translates \texttt{CART\_PASS} events into Model Context Protocol (MCP) commands.
		\item \textbf{Execution:} Facilitates AI-driven world-state manipulation (e.g., automated schematic injection).
	\end{itemize}
\end{frame}

% Slide 9: Safe Schematic Deployment
\begin{frame}{Safe Schematic Deployment}
	\begin{itemize}
		\item \textbf{Challenge:} Preventing AI-generated structures from overwriting player zones or telemetry transit lines.
		\item \textbf{Solution:} Spatial intersection middleware.
		\item \textbf{Validation:} Pre-deployment checks ensure NBT integrity and correct origin coordinates.
		\item \textbf{Result:} Eliminates chunk corruption and "silent deployment failures."
	\end{itemize}
\end{frame}

% Slide 10: Codebase Organization
\begin{frame}{Managing the Distributed Codebase}
	\begin{itemize}
		\item \textbf{Structure:} Server-centric organization (e.g., \texttt{src/servers/chonk/}, \texttt{src/servers/stargate/}).
		\item \textbf{Isolation:} Hardware-specific dependencies are strictly siloed.
		\item \textbf{Efficiency:} Hailo/Coral ML libraries remain entirely off the JVM-focused Chonk host.
		\item \textbf{Security:} Reduces the attack surface and enables targeted deployment scripts.
	\end{itemize}
\end{frame}

% Slide 11: Hardware Acceleration Roadmap
\begin{frame}{Hardware Acceleration Roadmap}
	\begin{itemize}
		\item \textbf{Objective:} Offload spatial parsing and visual inference from standard CPUs.
		\item \textbf{Edge TPUs:} Utilizing Coral Edge TPUs via the Mendel Development Tool (MDT).
		\item \textbf{NPUs:} Deploying compiled \texttt{.hef} models to Hailo architecture.
		\item \textbf{Impact:} Rapid object detection and schematic synthesis without impacting the live game simulation.
	\end{itemize}
\end{frame}
